const pages = dv.pages('"生产数据"').where(p => p.班组);
const teams = ["甲班", "乙班", "丙班", "丁班"];

const summary = [];

teams.forEach(team => {
  const rows = pages.where(p => p.班组 === team);
  if (!rows || rows.length === 0) return;

  let 蒸汽合计 = 0, 糖浆合计 = 0, 水合计 = 0, 电合计 = 0;
  let 正常班数 = 0, 水分扣分合计 = 0;

  rows.forEach(p => {
    const hasData = p.蒸汽消耗 !== undefined && p.蒸汽消耗 !== null && !isNaN(p.蒸汽消耗);
    if (hasData) {
      蒸汽合计 += Number(p.蒸汽消耗);
      糖浆合计 += Number(p.糖浆加量);
      水合计   += Number(p.水消耗);
      电合计   += Number(p.电消耗);
      正常班数 += 1;
    }

    const m = p.水分;
    if (m !== undefined && m !== null && !isNaN(m)) {
      if (m < 2)  水分扣分合计 -= 5;
      if (m > 2)  水分扣分合计 -= 10;
//      if (m < 10.5) 水分扣分合计 -= 5;
//      if (m > 11.5) 水分扣分合计 -= 10;
    }
  });

  const 蒸汽糖浆比 = 糖浆合计 ? (蒸汽合计 / 糖浆合计).toFixed(4) : "-";
  const 水平均 = 正常班数 ? (水合计 / 正常班数).toFixed(2) : "-";
  const 电平均 = 正常班数 ? (电合计 / 正常班数).toFixed(1) : "-";

  summary.push([
    team,
    正常班数,
    蒸汽合计,
    糖浆合计,
    水合计.toFixed(1),
    电合计,
    水分扣分合计,
//    蒸汽糖浆比,
//    水平均,
//    电平均
  ]);
});

//dv.header(2, "📊 各班小计与平均");
dv.header(2, "📊 各班小计");
dv.table(
//  ["班组","正常班数","蒸汽合计","糖浆合计","水合计","电合计","水分扣分","蒸汽÷糖浆","水平均","电平均"],
  ["班组","正常班数","蒸汽合计","糖浆合计","水合计","电合计","水分扣分"],
  summary
);
