const dataFolder = "生产数据";
const teams = ["甲班", "乙班", "丙班", "丁班"];
const processBaseScore = 40;

function cleanNumber(v) {
  if (v === undefined || v === null) return null;
  if (typeof v === "number") return v;
  if (typeof v !== "string") return null;
  let s = v.trim()
    .replace(/^["'\u2018\u2019"'"'"]+|["'\u2018\u2019"'"'"]+$/g, "")
    .replace(/,/g, "")
    .replace(/，/g, "");
  let m = s.match(/^-?\d+(\.\d+)?/);
  if (m) {
    let n = parseFloat(m[0]);
    return isNaN(n) ? null : n;
  }
  return null;
}

function safeNum(p, key) {
  let v = cleanNumber(p[key]);
  return v !== null ? v : 0;
}

function hasAnyData(p) {
  const keys = ["蒸汽消耗", "糖浆加量", "水消耗", "电消耗"];
  return keys.some(k => cleanNumber(p[k]) !== null);
}

function formatNum(v) {
  if (v === undefined || v === null || v === "" || v === "-") return "-";
  let n = parseFloat(v);
  if (isNaN(n)) return String(v);
  if (n === Math.floor(n)) return String(Math.floor(n));
  return String(n);
}

function findRepeatingDecimal(numerator, denominator, maxDigits = 30) {
  if (denominator === 0) return [null, null, null, false];

  let sign = "";
  if ((numerator < 0) !== (denominator < 0)) sign = "-";
  numerator = Math.abs(numerator);
  denominator = Math.abs(denominator);

  let integerPart = Math.floor(numerator / denominator);
  let remainder = numerator % denominator;

  if (remainder === 0) {
    return [sign + String(integerPart), "", "", false];
  }

  let remainders = {};
  let decimalDigits = [];
  let pos = 0;

  while (remainder !== 0 && pos < maxDigits) {
    if (remainders[remainder] !== undefined) {
      let start = remainders[remainder];
      let nonRepeating = decimalDigits.slice(0, start).join("");
      let repeating = decimalDigits.slice(start).join("");
      return [sign + String(integerPart), nonRepeating, repeating, true];
    }

    remainders[remainder] = pos;
    remainder *= 10;
    let digit = Math.floor(remainder / denominator);
    decimalDigits.push(String(digit));
    remainder = remainder % denominator;
    pos++;
  }

  if (remainder === 0) {
    let nonRepeating = decimalDigits.join("").replace(/0+$/, "");
    return [sign + String(integerPart), nonRepeating, "", false];
  } else {
    let nonRepeating = decimalDigits.join("");
    return [sign + String(integerPart), nonRepeating, "", false];
  }
}

function smartDivide(numerator, denominator, maxDigits = 4) {
  if (denominator === 0 || denominator === null || denominator === undefined) {
    return "-";
  }

  let [intPart, nonRep, rep, isRep] = findRepeatingDecimal(numerator, denominator, 30);

  if (intPart === null) return "-";

  if (isRep) {
    // 循环小数：循环节重复，保证总位数 > maxDigits + 1
    let minTotal = maxDigits + 2;  // 至少6位
    let repeatCount = Math.max(2, Math.ceil((minTotal - nonRep.length) / rep.length));

    if (nonRep) {
      let decimalPart = nonRep + rep.repeat(repeatCount);
      return intPart + "." + decimalPart;
    } else {
      let decimalPart = rep.repeat(repeatCount);
      return intPart + "." + decimalPart;
    }
  }

  // 能除尽（有限小数）
  if (nonRep === "") {
    return intPart;
  } else {
    return intPart + "." + nonRep;
  }
}

function smartFormat(value, maxDigits = 2) {
  if (value === undefined || value === null || value === "" || value === "-") {
    return "-";
  }
  let n = parseFloat(value);
  if (isNaN(n)) return String(value);
  if (n === Math.floor(n)) return String(Math.floor(n));

  let s = String(n);
  if (s.toLowerCase().includes('e')) {
    return String(Math.round(n * Math.pow(10, maxDigits + 2)) / Math.pow(10, maxDigits + 2));
  }

  if (s.includes('.')) {
    let decimalPart = s.split('.')[1];
    if (decimalPart.replace(/0/g, '') === '') {
      return String(Math.floor(n));
    }
  }

  return s;
}

function getWaterStatus(val) {
  if (val <= 5) {
    if (val === 2) return ["👌", 0];
    if (val < 2) return ["🍂", -5];
    return ["💦", -10];
  } else {
    if (val >= 10.5 && val <= 11.5) return ["👌", 0];
    if (val < 10.5) return ["🍂", -5];
    return ["💦", -10];
  }
}

function formatWater(m) {
  let val = cleanNumber(m);
  if (val === null) return "-";
  let [emoji, score] = getWaterStatus(val);
  return emoji;
}

function calcWaterScore(m) {
  let val = cleanNumber(m);
  if (val === null) return 0;
  let [emoji, score] = getWaterStatus(val);
  return score;
}

function rank(arr, key, asc=true) {
  const sorted = [...arr].sort((a,b) => {
    const av = a[1][key], bv = b[1][key];
    const aok = typeof av === "number", bok = typeof bv === "number";
    if (!aok && !bok) return 0;
    if (!aok) return 1;
    if (!bok) return -1;
    return asc ? av - bv : bv - av;
  });
  const vals = sorted.filter(x => typeof x[1][key] === "number").map(x => x[1][key]);
  const unique = [...new Set(vals)].sort((a,b) => asc ? a - b : b - a);
  const map = {};
  unique.forEach((v, i) => map[v] = i + 1);
  return map;
}

  return dv.pages(`"${dataFolder}"`).where(p => p.班组);

function calcTeamStats(pages) {
  const teamStats = {};
  const allRepairs = [];

  teams.forEach(team => {
    const rows = pages.where(p => p.班组 === team).sort(p => p.日期).array();
    if (rows.length === 0) return;

    let 蒸汽合计 = 0, 糖浆合计 = 0, 水合计 = 0, 电合计 = 0;
    let 正常班数 = 0, 水分扣分 = 0, 合格数 = 0;
    let 检维修数 = 0;
    const 明细 = [];

    rows.forEach(p => {
      const isRepair = p.类型 === "检维修";
      const hasData = hasAnyData(p) && !isRepair;

      if (hasData) {
        蒸汽合计 += safeNum(p, "蒸汽消耗");
        糖浆合计 += safeNum(p, "糖浆加量");
        水合计   += safeNum(p, "水消耗");
        电合计   += safeNum(p, "电消耗");
        正常班数 += 1;
      }

      if (isRepair) {
        检维修数 += 1;
      }

      const m = p.水分;
      const 本条扣分 = calcWaterScore(m);
      水分扣分 += 本条扣分;

      if (hasData && 本条扣分 === 0) {
        合格数 += 1;
      }

      if (isRepair) {
        allRepairs.push({ 班组: team, 日期: p.日期 || "-" });
      }

      if (isRepair) {
        明细.push({
          日期: p.日期 || "-",
          类型: "🛠",
          蒸汽: "-",
          糖浆: "-",
          水: "-",
          电: "-",
          水分: formatWater(p.水分)
        });
      } else {
        明细.push({
          日期: p.日期 || "-",
          类型: "✅",
          蒸汽: formatNum(p.蒸汽消耗),
          糖浆: formatNum(p.糖浆加量),
          水: formatNum(p.水消耗),
          电: formatNum(p.电消耗),
          水分: formatWater(p.水分)
        });
      }
    });

    // 使用智能除法
    const 蒸汽糖浆比 = smartDivide(蒸汽合计, 糖浆合计, 4);
    const 水平均 = smartDivide(水合计, 正常班数, 2);
    const 电平均 = smartDivide(电合计, 正常班数, 1);
    const 工艺分 = 正常班数 ? smartDivide(processBaseScore * 合格数, 正常班数, 2) : 0;

    teamStats[team] = {
      蒸汽: 蒸汽合计, 糖浆: 糖浆合计, 水: 水合计, 电: 电合计,
      扣分: 水分扣分, 班数: 正常班数, 检维修数: 检维修数, 合格数: 合格数, 工艺分: 工艺分,
      比: 蒸汽糖浆比, 水均: 水平均, 电均: 电平均,
      明细: 明细
    };
  });

  return { teamStats, allRepairs };
}

function calcRankings(teamStats) {
  const statsList = teams.map(t => [t, teamStats[t]]).filter(x => x[1]);

  const rRatio = rank(statsList, "比", true);
  const rWater = rank(statsList, "水均", true);
  const rElec  = rank(statsList, "电均", true);

  statsList.forEach(([t, s]) => {
    s.排名比 = rRatio[s.比] || "-";
    s.排名水 = rWater[s.水均] || "-";
    s.排名电 = rElec[s.电均] || "-";
    s.积分 = (typeof s.排名比 === "number" ? s.排名比 : 0) +
             (typeof s.排名水 === "number" ? s.排名水 : 0) +
             (typeof s.排名电 === "number" ? s.排名电 : 0);
  });

  const rScore = rank(statsList, "积分", true);
  statsList.forEach(([t, s]) => {
    s.总排名 = rScore[s.积分] || "-";
  });

  return statsList;
}

function renderLegend() {
  dv.header(2, "📖 图例说明");
  dv.table(
    ["符号", "含义", "说明"],
    [
      ["✅", "正常生产", "正常班次记录"],
      ["🛠", "检维修", "设备检修/维护班次"],
      ["👌", "水分达标", "水分值在合格范围内"],
      ["🍂", "水分偏低", "水分低于下限，扣5分"],
      ["💦", "水分偏高", "水分高于上限，扣10分"]
    ]
  );
}


// ═══════════════════════════════════════════════════════
// 第八部分：报表渲染

function renderSummaryTable(statsList) {
  dv.header(2, "📊 消耗统计汇总");
  dv.table(
    ["班组","蒸汽用量","糖浆加量","水量","电量","水分扣分"],
    statsList.map(([t, s]) => [t, formatNum(s.蒸汽), formatNum(s.糖浆), formatNum(s.水), formatNum(s.电), s.扣分])
  );
}

function renderAverageTable(statsList) {
  dv.header(2, "📈 平均分");
  dv.table(
    ["班组","蒸汽÷糖浆","水量÷正常班","电量÷正常班","工艺分"],
    statsList.map(([t, s]) => [t, s.比, formatNum(s.水均), formatNum(s.电均), s.工艺分])
  );
}

function renderRankingTable(statsList) {
  dv.header(2, "🏆 积分排名");
  dv.table(
    ["班组","蒸汽÷糖浆=排名","水消耗排名","电消耗排名","各班积分","最低消耗排名"],
    statsList.map(([t, s]) => [t, s.排名比, s.排名水, s.排名电, s.积分, s.总排名])
  );
}

function renderRepairTable(allRepairs) {
  if (allRepairs.length === 0) return;
  dv.header(2, "🔧 检维修记录");
  dv.table(
    ["班组","日期"],
    allRepairs.map(r => [r.班组, r.日期])
  );
}

function renderDetailTables(teamStats) {
  teams.forEach(team => {
    const s = teamStats[team];
    if (!s) return;

    const total = s.班数 + s.检维修数;

    dv.header(3, "📋 " + team + " 明细（" + total + "）");
    dv.table(
      ["日期","类型","蒸汽消耗","糖浆加量","水消耗","电消耗","水分"],
      s.明细.map(d => [d.日期, d.类型, d.蒸汽, d.糖浆, d.水, d.电, d.水分]).concat([
        ["**小计**", "✅: " + s.班数 + " | 🛠: " + s.检维修数, formatNum(s.蒸汽), formatNum(s.糖浆), formatNum(s.水), formatNum(s.电), s.扣分]
      ])
    );
  });
}


// ═══════════════════════════════════════════════════════
// 第九部分：主程序
// ═══════════════════════════════════════════════════════

function main() {
  const pages = loadPages();
  const { teamStats, allRepairs } = calcTeamStats(pages);
  const statsList = calcRankings(teamStats);

  renderLegend();
  renderSummaryTable(statsList);
  renderAverageTable(statsList);
  renderRankingTable(statsList);
  renderRepairTable(allRepairs);
  renderDetailTables(teamStats);
}

main();
