// 读取生产数据文件夹下的所有笔记
const pages = dv.pages('"生产数据"').where(p => p.班组);

// 按班组累加
const teamStats = {};
pages.forEach(p => {
  const team = p.班组;
  if (!teamStats[team]) {
    teamStats[team] = { 蒸汽:0, 糖浆:0, 水:0, 电:0, 正常班数:0, 水分扣分:0 };
  }
  
  // 有消耗数据的才算"正常班"
  const hasData = p.蒸汽消耗 !== undefined && p.蒸汽消耗 !== null && !isNaN(p.蒸汽消耗);
  if (hasData) {
    teamStats[team].蒸汽   += Number(p.蒸汽消耗);
    teamStats[team].糖浆   += Number(p.糖浆加量);
    teamStats[team].水     += Number(p.水消耗);
    teamStats[team].电     += Number(p.电消耗);
    teamStats[team].正常班数 += 1;
  }
  
  // 水分扣分（两个标准都写进来了，按需留一个）
  const m = p.水分;
  if (m !== undefined && m !== null && !isNaN(m)) {
    if (m < 2)  teamStats[team].水分扣分 -= 5;
    if (m > 2)  teamStats[team].水分扣分 -= 10;
//    if (m < 10.5) teamStats[team].水分扣分 -= 5;
 //   if (m > 11.5) teamStats[team].水分扣分 -= 10;
  }
});

// 组装数组
const teams = ["甲班","乙班","丙班","丁班"];
const stats = [];
teams.forEach(team => {
  const s = teamStats[team];
  if (!s) return;
  stats.push({
    班组: team,
    蒸汽用量: s.蒸汽,
    糖浆加量: s.糖浆,
    水量: s.水,
    电量: s.电,
    水分扣分: s.水分扣分,
    正常班数: s.正常班数,
    蒸汽糖浆比: s.糖浆 ? s.蒸汽/s.糖浆 : 0,
    水量平均: s.正常班数 ? s.水/s.正常班数 : 0,
    电量平均: s.正常班数 ? s.电/s.正常班数 : 0,
    工艺分: 40
  });
});

// 密集排名（并列同一名次，下一名次连续）
function denseRank(arr, key, asc=true) {
  const sorted = [...arr].sort((a,b) => asc ? a[key]-b[key] : b[key]-a[key]);
  const map = {};
  let rank = 1;
  for (let i=0; i<sorted.length; i++) {
    const v = sorted[i][key];
    if (i>0 && v !== sorted[i-1][key]) rank++;
    if (map[v] === undefined) map[v] = rank;
  }
  return map;
}

const rRatio = denseRank(stats, "蒸汽糖浆比", true);
const rWater = denseRank(stats, "水量平均", true);
const rElec  = denseRank(stats, "电量平均", true);

stats.forEach(s => {
  s.蒸汽糖浆排名 = rRatio[s.蒸汽糖浆比];
  s.水消耗排名   = rWater[s.水量平均];
  s.电消耗排名   = rElec[s.电量平均];
  s.各班积分     = s.蒸汽糖浆排名 + s.水消耗排名 + s.电消耗排名;
});

const rScore = denseRank(stats, "各班积分", true);
stats.forEach(s => s.最低消耗排名 = rScore[s.各班积分]);

// ========== 输出三张表 ==========

dv.header(2, "📊 消耗统计汇总");
dv.table(
  ["班组","蒸汽用量","糖浆加量","水量","电量","水分扣分"],
  stats.map(s => [s.班组, s.蒸汽用量, s.糖浆加量, s.水量.toFixed(1), s.电量, s.水分扣分])
);

dv.header(2, "📈 平均分");
dv.table(
  ["班组","蒸汽÷糖浆","水量÷正常班","电量÷正常班","工艺分"],
  stats.map(s => [
    s.班组,
    s.蒸汽糖浆比.toFixed(13),
    s.水量平均.toFixed(2),
    s.电量平均.toFixed(1),
    s.工艺分
  ])
);

dv.header(2, "🏆 积分排名");
dv.table(
  ["班组","蒸汽÷糖浆=排名","水消耗排名","电消耗排名","各班积分","最低消耗排名"],
  stats.map(s => [
    s.班组,
    s.蒸汽糖浆排名,
    s.水消耗排名,
    s.电消耗排名,
    s.各班积分,
    s.最低消耗排名
  ])
);

dv.header(2, "🔧 检维修记录");
const repairs = pages.where(p => p.类型 === "检维修");
dv.table(
  ["班组","日期","水分","扣分"],
  repairs.map(p => {
    let 扣分 = 0;
    const m = p.水分;
    if (m < 2) 扣分 -= 5;
    if (m > 2) 扣分 -= 10;
    return [p.班组, p.日期, p.水分, 扣分];
  })
);
